665  subroutine calc_soil_params(num_soil_cats, smlow, smhigh, satdk, &
666  maxsmc, bb, satpsi, satdw, f11, refsmc, drysmc, wltsmc)
667 
668  implicit none
669 
670  integer, intent(in) :: num_soil_cats
671 
672  real, intent(in) :: smlow, smhigh
673  real, intent(in) :: bb(num_soil_cats)
674  real, intent(in) :: maxsmc(num_soil_cats)
675  real, intent(in) :: satdk(num_soil_cats)
676  real, intent(in) :: satpsi(num_soil_cats)
677 
678  real, intent(out) :: f11(num_soil_cats)
679  real, intent(out) :: satdw(num_soil_cats)
680  real, intent(out) :: refsmc(num_soil_cats)

682  real, intent(out) :: wltsmc(num_soil_cats)
683 
684  integer :: i
685 
686  real :: refsmc1
687  real :: wltsmc1
688 
689  satdw = 0.0
690  f11 = 0.0
691  refsmc = 0.0
692  wltsmc = 0.0
693  drysmc = 0.0
694 
695  do i = 1, num_soil_cats
696 
697  if (maxsmc(i) > 0.0) then
698 
699  satdw(i) = bb(i)*satdk(i)*(satpsi(i)/maxsmc(i))
700  f11(i) = alog10(satpsi(i)) + bb(i)*alog10(maxsmc(i)) + 2.0
