60  SUBROUTINE skgb2(LUGB,ISEEK,MSEEK,LSKIP,LGRIB,I1)

63 ! subprogram: skgb2
64 ! prgmmr: gayno org: w/np2 date: 2014-feb-07
65 !
66 ! abstract: determine whether file is grib or not.
67 ! based on w3nco library routine skgb.
68 !
69 ! program history log:
70 ! 2014-feb-07 gayno - initial version
71 !
72 ! usage: call SKGB2(LUGB,ISEEK,MSEEK,LSKIP,LGRIB,I1)
73 !
74 ! input argument list: lugb - file unit number
75 ! iseek - number of bits to skip
76 ! before search.
77 ! mseek - max number of bytes
78 ! to search.
79 !
80 ! output argument list: lskip - number of bytes to skip
81 ! before message
82 ! lgrib - number of bytes in message.
83 ! '0' if not grib.
84 ! i1 - '1' or '2' if grib1/2 file.
85 ! '0' if not grib.
86 !
87 ! files:
88 ! input:
89 ! - file to be checked, unit=lugb
90 !
91 ! output: none
92 !
93 ! condition codes: none
94 !
95 ! remarks: none.
96 !
97 !$$$
98  implicit none
99  INTEGER, INTENT( IN) :: lugb, iseek, mseek
100  INTEGER, INTENT(OUT) :: lskip, lgrib, i1
101  INTEGER, PARAMETER :: lseek=128
102  INTEGER :: k, kz, ks, kg, kn, km, i4, k4
103  CHARACTER z(lseek)
104  CHARACTER z4(4)
105 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
106  i1=0
107  lgrib=0
108  ks=iseek
109  kn=min(lseek,mseek)
110  kz=lseek
111 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
112 ! LOOP UNTIL GRIB MESSAGE IS FOUND
113  DO WHILE(lgrib.EQ.0.AND.kn.GE.8.AND.kz.EQ.lseek)
114 ! READ PARTIAL SECTION
115  CALL baread(lugb,ks,kn,kz,z)
116  km=kz-8+1
117  k=0
118 ! LOOK FOR 'GRIB...1' IN PARTIAL SECTION
119  DO WHILE(lgrib.EQ.0.AND.k.LT.km)
120  CALL gbytec(z,i4,(k+0)*8,4*8)
121  CALL gbytec(z,i1,(k+7)*8,1*8)
122  IF(i4.EQ.1196575042.AND.(i1.EQ.1.OR.i1.EQ.2)) THEN
123 ! LOOK FOR '7777' AT END OF GRIB MESSAGE
124  IF (i1.EQ.1) CALL gbytec(z,kg,(k+4)*8,3*8)
125  IF (i1.EQ.2) CALL gbytec(z,kg,(k+12)*8,4*8)
126  CALL baread(lugb,ks+k+kg-4,4,k4,z4)
127  IF(k4.EQ.4) THEN
128  CALL gbytec(z4,i4,0,4*8)
129  IF(i4.EQ.926365495) THEN
130 ! GRIB MESSAGE FOUND
131  lskip=ks+k
132  lgrib=kg
133  ENDIF
134  ENDIF
135  ENDIF
136  k=k+1
137  ENDDO
138  ks=ks+km
139  kn=min(lseek,iseek+mseek-ks)
140  ENDDO
141 ! - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -
142  RETURN
143  END subroutine skgb2
