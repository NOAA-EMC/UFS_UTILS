482  grid[n].latt1D = (double *) malloc(ny[n]*sizeof(double));

484  grid[n].latc1D = (double *) malloc((ny[n]+1)*sizeof(double));

486  y = (double *) malloc((2*nx[n]+1)*(2*ny[n]+1)*sizeof(double));

493  ii = 2*(i + grid[n].isc);

495  grid[n].latc[j*(grid[n].nxc+1)+i] = y[jj*(2*nx[n]+1)+ii] * D2R;
496  }
497 
498  if(grid[n].is_tripolar)
499  ind = nx[n]/4;
500  else
501  ind = 0;
502  for(i=0; i<nx[n]+1; i++) grid[n].lonc1D[i] = x[2*i] * D2R;
503  for(j=0; j<ny[n]+1; j++) grid[n].latc1D[j] = y[2*j*(2*nx[n]+1)+2*ind] * D2R;
504  if(!grid[n].is_tripolar) ind = 1;
505

507  for(j=0; j<ny[n]; j++) grid[n].latt1D[j] = y[(2*j+1)*(2*nx[n]+1) + 2*ind] * D2R;
508  free(x);
509  free(y);

511  /* if vector, need to get rotation angle */
512  /* we assume the grid is orthogonal */
513  if(opcode & VECTOR) {
514  if(opcode & AGRID) {
515  double *angle;

517  grid[n].cosrot = (double *) malloc(nx[n]*ny[n]*sizeof(double));
518  grid[n].sinrot = (double *) malloc(nx[n]*ny[n]*sizeof(double));
519  vid = mpp_get_varid(g_fid, "angle_dx");
520  mpp_get_var_value(g_fid, vid, angle);
521  grid[n].rotate = 0;
522  for(j=0; j<grid[n].nyc; j++) for(i=0; i<grid[n].nxc; i++) {
523  jj = 2*(j + grid[n].jsc);
524  ii = 2*(i + grid[n].isc);

526  grid[n].sinrot[j*grid[n].nxc+i] = sin(angle[jj*(2*nx[n]+1)+ii]*D2R);
527  if(fabs(grid[n].sinrot[j*grid[n].nxc+i]) > EPSLN10) grid[n].rotate = 1;

530  }

609  dlat=lat_range/ny_fine;
610  for(j=0; j<ny_fine; j++)grid->latt1D_fine[j] = (latbegin+(j+0.5)*dlat)*D2R;
611  }

613  dlat = lat_range/(ny_fine-1);
614  for(j=0; j<ny_fine; j++) grid->latt1D_fine[j] = (latbegin+j*dlat)*D2R;

616  }

620  /* get the cartesian coordinates */
621  latlon2xyz(nx_fine*ny_fine, grid->lont, grid->latt, grid->xt, grid->yt, grid->zt);
622 
623  unit_vect_latlon(nx_fine*ny_fine, grid->lont, grid->latt, grid->vlon_t, grid->vlat_t);
624 
625  }
626 
627  grid->lonc = (double *) malloc((nxc+1)*(nyc+1)*sizeof(double));
628  grid->latc = (double *) malloc((nxc+1)*(nyc+1)*sizeof(double));
629  for(j=0; j<=nyc; j++) {
630  jj = j + grid->jsc;
631  for(i=0; i<=nxc; i++) {
632  ii = i + grid->isc;
633  grid->lonc[j*(nxc+1)+i] = grid->lonc1D[ii];
634  grid->latc[j*(nxc+1)+i] = grid->latc1D[jj];
635  }
636  }
637  if(opcode & VECTOR) { /* no rotation is needed for regular lat-lon grid. */
638  grid->rotate = 0;
639  }
